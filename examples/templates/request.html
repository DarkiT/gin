代码示例

框架提供简单直观的API，轻松获取请求的各种信息。

// 获取基本请求信息
method := c.Method() // 获取请求方法 (GET, POST, PUT等)
path := c.Path() // 获取请求路径
fullUrl := c.FullUrl() // 获取完整URL
host := c.Host() // 获取主机名
domain := c.Domain() // 获取域名
scheme := c.Scheme() // 获取协议 (http或https)
port := c.Port() // 获取端口号
userAgent := c.UserAgent()// 获取用户代理
ip := c.ClientIP() // 获取客户端IP地址
// 判断请求特性
isSSL := c.IsSsl() // 判断是否是HTTPS请求
isAjax := c.IsAjax() // 判断是否是AJAX请求
isJson := c.IsJson() // 判断是否请求JSON数据
isForm := c.IsForm() // 判断是否表单请求
isPjax := c.IsPjax() // 判断是否PJAX请求
// 获取请求头和Cookie
header := c.Header("X-Custom-Header") // 获取指定请求头
headers := c.Headers() // 获取所有请求头
cookie := c.Cookie("session") // 获取指定Cookie
cookies := c.Cookies() // 获取所有Cookie
// 获取请求参数
query := c.Query("name") // 获取查询参数
queryDefault := c.DefaultQuery("page", "1") // 带默认值的查询参数
param := c.Param("id") // 获取路由参数
form := c.PostForm("username") // 获取表单参数
formDefault := c.DefaultPostForm("age", "18") // 带默认值的表单参数
