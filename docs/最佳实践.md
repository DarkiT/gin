### 分层架构示例

```go
// models/user.go - 数据模型
package models

type User struct {
    ID    int    `json:"id"`
    Name  string `json:"name" binding:"required"`
    Email string `json:"email" binding:"required,email"`
}

// repository/user_repo.go - 数据访问层
package repository

type UserRepository interface {
    FindAll() ([]models.User, error)
    FindByID(id int) (*models.User, error)
    Create(user *models.User) error
    Update(user *models.User) error
    Delete(id int) error
}

type userRepo struct {
    db *sql.DB
}

func NewUserRepository(db *sql.DB) UserRepository {
    return &userRepo{db: db}
}

// service/user_service.go - 业务逻辑层
package service

type UserService interface {
    GetUsers() ([]models.User, error)
    GetUser(id int) (*models.User, error)
    CreateUser(user *models.User) error
}

type userService struct {
    repo repository.UserRepository
}

func NewUserService(repo repository.UserRepository) UserService {
    return &userService{repo: repo}
}

func (s *userService) GetUsers() ([]models.User, error) {
    return s.repo.FindAll()
}

// handlers/user.go - API 处理器
package handlers

type UserHandler struct {
    service service.UserService
}

func NewUserHandler(service service.UserService) *UserHandler {
    return &UserHandler{service: service}
}

func (h *UserHandler) ListUsers(c *Context) {
    users, err := h.service.GetUsers()
    if err != nil {
        c.ServerError("获取用户列表失败")
        return
    }
    c.Success(users)
}

// cmd/server/main.go - 主入口
package main

func main() {
    // 初始化依赖
    db := initDB()
    userRepo := repository.NewUserRepository(db)
    userService := service.NewUserService(userRepo)
    userHandler := handlers.NewUserHandler(userService)

    // 创建路由
    router := gin.Default()

    api := router.Group("/api")
    {
        api.GET("/users", userHandler.ListUsers)
    }

    router.Run(":8080")
}
```

## ⚠️ 错误处理

### ✅ 使用统一的错误响应

```go
// ✅ 好的做法
router.GET("/users/:id", func(c *gin.Context) {
    id := c.ParamInt("id")

    user, err := userService.GetUser(id)
    if err != nil {
        if errors.Is(err, ErrNotFound) {
            c.NotFound("用户不存在")  // 明确的错误类型
        } else {
            c.ServerError("获取用户失败")
        }
        return
    }

    c.Success(user)
})

// ❌ 不好的做法
router.GET("/users/:id", func(c *gin.Context) {
    id := c.ParamInt("id")
    user, _ := userService.GetUser(id)  // 忽略错误
    c.JSON(200, user)  // 不统一的响应格式
})
```

### ✅ 定义业务错误

```go
// 定义错误类型
var (
    ErrNotFound      = errors.New("资源不存在")
    ErrUnauthorized  = errors.New("未授权")
    ErrInvalidInput  = errors.New("输入无效")
)

// 使用自定义错误
func (s *userService) GetUser(id int) (*User, error) {
    user, err := s.repo.FindByID(id)
    if err != nil {
        if err == sql.ErrNoRows {
            return nil, ErrNotFound
        }
        return nil, fmt.Errorf("查询用户失败: %w", err)
    }
    return user, nil
}
```

### ✅ 错误日志记录

```go
func (h *UserHandler) CreateUser(c *gin.Context) {
    var user models.User
    if !c.BindJSON(&user) {
        return
    }

    err := h.service.CreateUser(&user)
    if err != nil {
        // 记录详细错误日志
        log.Printf("创建用户失败: %v, 请求ID: %s", err, c.GetString("request_id"))

        // 返回用户友好的错误信息
        c.ServerError("创建用户失败")
        return
    }

    c.Created(user)
}
```

## ⚡ 性能优化

### ✅ 使用缓存

```go
// 缓存热点数据
router.GET("/users/:id", func(c *gin.Context) {
    id := c.Param("id")
    cacheKey := "user:" + id

    cache := c.GetCache()

    // 尝试从缓存获取
    if cachedUser, found := cache.Get(cacheKey); found {
        c.Success(cachedUser)
        return
    }

    // 从数据库获取
    user, err := userService.GetUser(id)
    if err != nil {
        c.ServerError("获取用户失败")
        return
    }

    // 缓存 5 分钟
    cache.Set(cacheKey, user, 5*time.Minute)

    c.Success(user)
})
```

```go
func initDB() *sql.DB {
    db, err := sql.Open("postgres", dsn)
```

### ✅ 并发处理

```go
// 并发获取多个资源
router.GET("/dashboard", func(c *gin.Context) {
    var (
        users    []User
        posts    []Post
        stats    Stats
        wg       sync.WaitGroup
        mu       sync.Mutex
        errors   []error
    )

    // 并发获取数据
    wg.Add(3)

    go func() {
        defer wg.Done()
        result, err := userService.GetUsers()
        mu.Lock()
        users = result
        if err != nil {
            errors = append(errors, err)
        }
        mu.Unlock()
    }()

    go func() {
        defer wg.Done()
        result, err := postService.GetPosts()
        mu.Lock()
        posts = result
        if err != nil {
            errors = append(errors, err)
        }
        mu.Unlock()
    }()

    go func() {
        defer wg.Done()
        result, err := statsService.GetStats()
        mu.Lock()
        stats = result
        if err != nil {
            errors = append(errors, err)
        }
        mu.Unlock()
    }()

    wg.Wait()

    if len(errors) > 0 {
        c.ServerError("获取数据失败")
        return
    }

    c.Success(gin.H{
        "users": users,
        "posts": posts,
        "stats": stats,
    })
})
```

```go
type CreateUserRequest struct {
    Name     string `json:"name" binding:"required,min=2,max=50"`
    Email    string `json:"email" binding:"required,email"`
    Password string `json:"password" binding:"required,min=8"`
    Age      int    `json:"age" binding:"gte=18,lte=120"`
}

router.POST("/users", func(c *gin.Context) {
    var req CreateUserRequest

    // 自动验证
    if !c.BindJSON(&req) {
        return  // 验证失败自动返回 400

    // 额外的业务验证
    if !isValidEmail(req.Email) {
        c.ValidationError(gin.H{"email": "邮箱格式不正确"})
        return
    }

    user := createUser(req)
    c.Created(user)
})
```

### ✅ 密码处理

```go
import "golang.org/x/crypto/bcrypt"

// 密码加密
func hashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(bytes), err
}

// 密码验证
func checkPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}

// 使用
router.POST("/register", func(c *gin.Context) {
    var req RegisterRequest
    if !c.BindJSON(&req) {
        return
    }

    // 加密密码
    hashedPassword, err := hashPassword(req.Password)
    if err != nil {
        c.ServerError("注册失败")
        return
    }

    user := User{
        Name:     req.Name,
        Email:    req.Email,
        Password: hashedPassword,  // 存储加密后的密码
    }

    createUser(user)
    c.Created(user)
})
```

### ✅ CORS 安全配置

```go
// ✅ 生产环境明确指定域名
router := gin.NewRouter(
    gin.WithCORS(
        "https://example.com",
        "https://app.example.com",
    ),
)

// ❌ 避免使用通配符
router := gin.NewRouter(
    gin.WithCORS("*"),  // 危险！允许所有域名
)

// ✅ 详细配置（自定义中间件）
router.Use(func(c *gin.Context) {
    origin := c.GetHeader("Origin")
    if origin != "https://example.com" {
        c.AbortWithStatus(http.StatusForbidden)
        return
    }

    c.Header("Access-Control-Allow-Origin", origin)
    c.Header("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE")
    c.Header("Access-Control-Allow-Headers", "Content-Type,Authorization")
    c.Header("Access-Control-Expose-Headers", "X-Total-Count")
    c.Header("Access-Control-Allow-Credentials", "true")
    if c.Request.Method == http.MethodOptions {
        c.AbortWithStatus(http.StatusNoContent)
        return
    }
    c.Next()
})
```

```go
// 启用限流
router := gin.NewRouter(
    gin.WithRateLimit(100),  // 100 次/分钟

// 自定义限流中间件
func RateLimitMiddleware(limit int) gin.HandlerFunc {
    limiter := rate.NewLimiter(rate.Every(time.Minute/time.Duration(limit)), limit)

    return func(c *gin.Context) {
        if !limiter.Allow() {
            c.Fail("请求过于频繁，请稍后再试")
            c.Abort()
            return
        }
        c.Next()
    }
}
```

```go
func main() {
    // 从环境变量读取配置
    ginMode := getEnv("GIN_MODE", "release")
    port := getEnv("PORT", "8080")
    jwtSecret := mustGetEnv("JWT_SECRET")

    router := gin.NewRouter(
        gin.WithGinMode(ginMode),
        gin.WithJWT(jwtSecret),
    )

    // 健康检查
    router.Health()

    // 启动服务器
    router.Run(":" + port)
}

func getEnv(key, defaultValue string) string {
    if value := os.Getenv(key); value != "" {
        return value
    }
    return defaultValue
}

func mustGetEnv(key string) string {
    value := os.Getenv(key)
    if value == "" {
        log.Fatalf("%s environment variable is required", key)
    }
    return value
}
```

### ✅ 优雅停机

```go
func main() {
    router := gin.Default()

    // ... 注册路由 ...

    // 创建 HTTP 服务器
    srv := &http.Server{
        Addr:    ":8080",
        Handler: router,
    }

    // 启动服务器（goroutine）
    go func() {
        if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
            log.Fatalf("启动失败: %v", err)
        }
    }()

    // 等待中断信号
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit

    log.Println("正在关闭服务器...")

    // 优雅停机（30秒超时）
    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()

    if err := srv.Shutdown(ctx); err != nil {
        log.Fatal("服务器强制关闭:", err)
    }

    log.Println("服务器已停止")
}
```
