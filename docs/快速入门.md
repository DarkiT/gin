## 🎯 第三步：RESTful API（10 分钟）

创建一个用户管理 API，包含 CRUD 操作。

### 3.1 定义数据模型

```go
// 用户模型
type User struct {
    ID    int    `json:"id"`
    Name  string `json:"name" binding:"required"`
    Email string `json:"email" binding:"required,email"`
}

// 模拟数据库
var (
    users   = make(map[int]User)
    nextID  = 1
    usersMu sync.RWMutex
)
```

### 3.2 实现处理函数

```go
// 获取用户列表
func listUsers(c *gin.Context) {
    usersMu.RLock()
    defer usersMu.RUnlock()

    list := make([]User, 0, len(users))
    for _, user := range users {
        list = append(list, user)
    }

    c.Success(list)  // ✨ 统一成功响应
}

// 获取单个用户
func getUser(c *gin.Context) {
    id, _ := strconv.Atoi(c.Param("id"))

    usersMu.RLock()
    user, exists := users[id]
    usersMu.RUnlock()

    if !exists {
        c.NotFound("用户不存在")  // ✨ 404 响应
        return
    }

    c.Success(user)
}

// 创建用户
func createUser(c *gin.Context) {
    var user User

    // 绑定并验证 JSON
    if !c.BindJSON(&user) {
        c.ValidationError(gin.H{"error": "数据格式错误"})  // ✨ 400 响应
        return
    }

    usersMu.Lock()
    user.ID = nextID
    nextID++
    users[user.ID] = user
    usersMu.Unlock()

    c.Created(user)  // ✨ 201 响应
}

// 更新用户
func updateUser(c *gin.Context) {
    id, _ := strconv.Atoi(c.Param("id"))

    usersMu.RLock()
    _, exists := users[id]
    usersMu.RUnlock()

    if !exists {
        c.NotFound("用户不存在")
        return
    }

    var user User
    if !c.BindJSON(&user) {
        c.ValidationError(gin.H{"error": "数据格式错误"})
        return
    }

    usersMu.Lock()
    user.ID = id
    users[id] = user
    usersMu.Unlock()

    c.Success(user)
}

// 删除用户
func deleteUser(c *gin.Context) {
    id, _ := strconv.Atoi(c.Param("id"))

    usersMu.Lock()
    delete(users, id)
    usersMu.Unlock()

    c.NoContent()  // ✨ 204 响应
}
```

### 3.3 注册路由

```go
func main() {
    router := gin.Default()

    // 健康检查
    router.Health()  // ✨ 自动添加 GET /health

    // API 路由组
    api := router.Group("/api")
    {
        // 用户 CRUD
        api.GET("/users", listUsers)
        api.GET("/users/:id", getUser)
        api.POST("/users", createUser)
        api.PUT("/users/:id", updateUser)
        api.DELETE("/users/:id", deleteUser)
    }

    router.Run(":8080")
}
```

### 3.4 测试 API

```bash
# 创建用户
curl -X POST http://localhost:8080/api/users \
  -H "Content-Type: application/json" \
  -d '{"name":"张三","email":"zhang@example.com"}'
# 响应: {"code":0,"message":"success","data":{"id":1,"name":"张三","email":"zhang@example.com"}}

# 获取用户列表
curl http://localhost:8080/api/users
# 响应: {"code":0,"message":"success","data":[{"id":1,"name":"张三","email":"zhang@example.com"}]}

# 获取单个用户
curl http://localhost:8080/api/users/1

# 更新用户
curl -X PUT http://localhost:8080/api/users/1 \
  -H "Content-Type: application/json" \
  -d '{"name":"张三（已更新）","email":"zhang@example.com"}'

# 删除用户
curl -X DELETE http://localhost:8080/api/users/1
# 响应: 204 No Content
```

**✅ 完成！你已经创建了一个完整的 RESTful API**

## 🔐 第四步：JWT 认证（5 分钟）

添加 JWT 认证保护 API。

### 4.1 创建登录接口

```go
// 登录请求
type LoginRequest struct {
    Username string `json:"username" binding:"required"`
    Password string `json:"password" binding:"required"`
}

// 登录处理
func login(c *gin.Context) {
    var req LoginRequest
    if !c.BindJSON(&req) {
        c.ValidationError(gin.H{"error": "登录信息格式错误"})
        return
    }

    // 简单验证（实际应查询数据库）
    if req.Username != "admin" || req.Password != "password" {
        c.Unauthorized("用户名或密码错误")  // ✨ 401 响应
        return
    }

    // 生成 JWT 令牌
    token, err := c.CreateJWTSession("your-secret-key", 2*time.Hour, gin.H{
        "username": req.Username,
        "role":     "admin",
    })

    if err != nil {
        c.ServerError("生成令牌失败")  // ✨ 500 响应
        return
    }

    c.Success(gin.H{
        "token": token,
        "expires_in": 7200,
    })
}
```

### 4.2 创建认证中间件

```go
// JWT 认证中间件
func AuthMiddleware(c *gin.Context) {
    jwt, ok := c.RequireJWT()  // ✨ 自动验证 JWT
    if !ok {
        return  // RequireJWT 已处理错误响应
    }

    // 设置用户信息到上下文
    c.Set("username", jwt["username"])
    c.Set("role", jwt["role"])

    c.Next()
}
```

### 4.3 应用认证

```go
func main() {
    router := gin.NewRouter(
        gin.WithJWT("your-secret-key"),  // ✨ 启用 JWT
    )

    // 公开路由
    router.POST("/login", login)

    // 受保护的路由
    api := router.Group("/api")
    api.Use(AuthMiddleware)  // 应用认证中间件
    {
        api.GET("/users", listUsers)
        api.POST("/users", createUser)
        // ... 其他路由
    }

    router.Run(":8080")
}
```

### 4.4 测试认证

```bash
# 1. 登录获取令牌
TOKEN=$(curl -X POST http://localhost:8080/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"password"}' \
  | jq -r '.data.token')

# 2. 使用令牌访问受保护 API
curl http://localhost:8080/api/users \
  -H "Authorization: Bearer $TOKEN"

# 3. 不带令牌访问（应失败）
curl http://localhost:8080/api/users
# 响应: {"code":401,"message":"未提供令牌"}
```

**✅ 完成！你的 API 现在有了 JWT 认证**

### 5.2 自定义中间件

```go
// 日志中间件
func LoggerMiddleware(c *gin.Context) {
    start := time.Now()
    path := c.Request.URL.Path

    c.Next()  // 执行后续处理

    latency := time.Since(start)
    status := c.Writer.Status()

    log.Printf("[%s] %s - %d - %v", c.Method(), path, status, latency)
}

// 应用中间件
router.Use(LoggerMiddleware)
```

## 💾 第六步：缓存系统（5 分钟）

### 6.1 启用缓存

```go
import "github.com/darkit/gin/cache"

router := gin.NewRouter(
    gin.WithCache(&cache.Config{
        TTL:             30 * time.Minute,   // 默认过期时间
        CleanupInterval: 5 * time.Minute,    // 清理间隔
    }),
)
```

### 6.2 使用缓存

```go
func getUser(c *gin.Context) {
    id := c.Param("id")
    cacheKey := "user:" + id

    // 尝试从缓存获取
    cache := c.GetCache()
    if cachedUser, found := cache.Get(cacheKey); found {
        c.Success(cachedUser)  // 返回缓存数据
        return
    }

    // 从"数据库"获取
    usersMu.RLock()
    user, exists := users[atoi(id)]
    usersMu.RUnlock()

    if !exists {
        c.NotFound("用户不存在")
        return
    }

    // 设置缓存（5分钟）
    cache.SetWithTTL(cacheKey, user, 5*time.Minute)

    c.Success(user)
}
```

```go
package main

import (
    "sync"
    "time"

    "github.com/darkit/gin"
    "github.com/darkit/gin/cache"
)

// ========== 数据模型 ==========

type User struct {
    ID    int    `json:"id"`
    Name  string `json:"name" binding:"required"`
    Email string `json:"email" binding:"required,email"`
}

type LoginRequest struct {
    Username string `json:"username" binding:"required"`
    Password string `json:"password" binding:"required"`
}

// ========== 模拟数据库 ==========

var (
    users   = make(map[int]User)
    nextID  = 1
    usersMu sync.RWMutex
)

// ========== 处理函数 ==========

func login(c *gin.Context) {
    var req LoginRequest
    if !c.BindJSON(&req) {
        return
    }

    if req.Username != "admin" || req.Password != "password" {
        c.Unauthorized("用户名或密码错误")
        return
    }

    token, _ := c.CreateJWTSession("your-secret-key", 2*time.Hour, gin.H{
        "username": req.Username,
    })

    c.Success(gin.H{"token": token})
}

func listUsers(c *gin.Context) {
    usersMu.RLock()
    defer usersMu.RUnlock()

    list := make([]User, 0, len(users))
    for _, user := range users {
        list = append(list, user)
    }

    c.Success(list)
}

func getUser(c *gin.Context) {
    id := c.ParamInt("id")

    // 缓存逻辑
    cacheKey := fmt.Sprintf("user:%d", id)
    cache := c.GetCache()

    if cachedUser, found := cache.Get(cacheKey); found {
        c.Success(cachedUser)
        return
    }

    usersMu.RLock()
    user, exists := users[id]
    usersMu.RUnlock()

    if !exists {
        c.NotFound("用户不存在")
        return
    }

    cache.SetWithTTL(cacheKey, user, 5*time.Minute)
    c.Success(user)
}

func createUser(c *gin.Context) {
    var user User
    if !c.BindJSON(&user) {
        return
    }

    usersMu.Lock()
    user.ID = nextID
    nextID++
    users[user.ID] = user
    usersMu.Unlock()

    c.Created(user)
}

func updateUser(c *gin.Context) {
    id := c.ParamInt("id")

    var user User
    if !c.BindJSON(&user) {
        return
    }

    usersMu.Lock()
    user.ID = id
    users[id] = user
    usersMu.Unlock()

    // 清除缓存
    c.GetCache().Delete(fmt.Sprintf("user:%d", id))

    c.Success(user)
}

func deleteUser(c *gin.Context) {
    id := c.ParamInt("id")

    usersMu.Lock()
    delete(users, id)
    usersMu.Unlock()

    c.GetCache().Delete(fmt.Sprintf("user:%d", id))
    c.NoContent()
}

// ========== 中间件 ==========

func AuthMiddleware(c *gin.Context) {
    jwt, ok := c.RequireJWT()
    if !ok {
        return
    }

    c.Set("username", jwt["username"])
    c.Next()
}

// ========== 主函数 ==========

func main() {
    // 创建路由器
    router := gin.NewRouter(
        gin.WithGinMode("debug"),
        gin.WithJWT("your-secret-key"),
        gin.WithCache(&cache.Config{
            TTL:             30 * time.Minute,
            CleanupInterval: 5 * time.Minute,
        }),
        gin.WithCORS("*"),
        gin.WithRequestID(),
    )

    // 健康检查
    router.Health()

    // 公开路由
    router.POST("/login", login)

    // 受保护的 API
    api := router.Group("/api")
    api.Use(AuthMiddleware)
    {
        api.GET("/users", listUsers)
        api.GET("/users/:id", getUser)
        api.POST("/users", createUser)
        api.PUT("/users/:id", updateUser)
        api.DELETE("/users/:id", deleteUser)
    }

    // 启动服务器
    router.Run(":8080")
}
```
